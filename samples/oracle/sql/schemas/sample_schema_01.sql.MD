###### Oracle PL/SQL

## 1. DDL Create

```sql
-- Department table
CREATE TABLE dept (
   deptno   NUMBER(2,0)    CONSTRAINT pk_dept PRIMARY KEY,
   dname    VARCHAR2(14),
   loc      VARCHAR2(13)
);

-- Employee table
CREATE TABLE emp (
   empno    NUMBER(4,0)    CONSTRAINT pk_emp PRIMARY KEY,
   ename    VARCHAR2(10),
   job      VARCHAR2(9),
   mgr      NUMBER(4,0),
   hiredate DATE,
   sal      NUMBER(7,2),
   comm     NUMBER(7,2),
   deptno   NUMBER(2,0)    CONSTRAINT fk_deptno REFERENCES dept(deptno)
);
```
